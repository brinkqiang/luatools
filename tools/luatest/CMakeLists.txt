cmake_minimum_required(VERSION 3.2)

PROJECT(luatest)
INCLUDE(cmake/ModuleImport.cmake)
INCLUDE(cmake/LuaModule.cmake)
LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
INCLUDE(cmake/ModuleCompileOptions.cmake)
ModuleSetCompileOptions()

SET(LUA_MODULE lua53)
SET(LUATEST_VERSION_MAJOR "1")
SET(LUATEST_VERSION_MINOR "0")
SET(LUATEST_VERSION_PATCH "1")
SET(LUATEST_VERSION "${LUATEST_VERSION_MAJOR}.${LUATEST_VERSION_MINOR}.${LUATEST_VERSION_PATCH}")

MESSAGE(STATUS "VERSION: ${LUATEST_VERSION}")

ADD_DEFINITIONS(-DUSE_LUATEST)
ADD_DEFINITIONS(-DLUA_COMPAT_MODULE -DLUA_COMPAT_APIINTCASTS)

LuaModuleImport("${LUA_MODULE}" "luatest" "src/luatest")
